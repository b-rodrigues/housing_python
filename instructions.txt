# Get python installed by any means on your computer
# you could install python using pyenv:
pyenv install 3.12.1

# switch to project directory and set up global python version
pyenv global 3.12.1

# Install pipenv
pip install pipenv --user

# install packages in your project's directory
pipenv install plotnine polars pandas beautifulsoup4

# how to install a package from github using pipenv  
pipenv install -e git+https://github.com/has2k1/plotnine.git@main#egg=plotnine
